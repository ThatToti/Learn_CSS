# CSS3多列布局

- columns: column-width|column-count

- column-width

  > - auto
  > - length

- column-count

  > - auto
  > - integer

- column-gap

  > - normal
  > - length

- column-rule

  > - column-rule-width
  >
  > - column-rule-style
  >
  > - column-rule-color
  >
  >   > 类似border的属性

- column-span

  > - none
  > - all

- column-fill

  > - auto
  >
  > - balance
  >
  >   > balance根据内容最多一列来，auto根据内容自己来
